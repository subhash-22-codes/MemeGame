// GameContext.tsx
import React, { createContext, useContext, useState, useEffect,useRef } from 'react';
import { io, Socket } from 'socket.io-client';
import { MEMES } from '../data/memes';
import { useAuth } from './AuthContext';

export type Player = {
  id: string;
  username: string;
  avatar?: string;
  score: number;
  isJudge: boolean;
  isReady: boolean;
  isHost?: boolean; // ✅ Add this to track if player is host
};

export type Meme = {
  id: string;
  url: string;
  title: string;
};

export type MemeSubmission = {
  playerId: string;
  memeId: string;
  score?: number;
};

export type GameState = {
  roomId: string;
  players: Player[];
  currentJudge?: Player;
  currentSentence?: string;
  roundNumber: number;
  totalRounds: number;
  roundsPerJudge: number;
  timeLeft: number;
  gamePhase:
    | 'lobby'
    | 'judgeSelection'
    | 'sentenceCreation'
    | 'memeSelection'
    | 'memeReveal'
    | 'scoring'
    | 'results';
  submissions: MemeSubmission[];
  availableMemes: Meme[];
};

export type ChatMessage = {
  id: string;
  playerId: string;
  playerName: string;
  message: string;
  timestamp: Date;
};

type GameContextType = {
  gameState: GameState | null;
  isHost: boolean;
  alreadyHost: boolean; // ✅ Add this
  setAlreadyHost: (value: boolean) => void;
  createRoom: (settings: { rounds: number; roundsPerJudge: number }) => Promise<string>;
  joinRoom: (roomId: string) => Promise<void>;
  leaveRoom: () => void;
  discardRoom: () => void; // <- Expose this
  startJudgeSelection: () => void;
  submitSentence: (sentence: string) => void;
  selectMeme: (memeId: string) => void;
  scoreMeme: (playerId: string, score: number) => void;
  startNextRound: () => void;
  endGame: () => void;
  chatMessages: ChatMessage[];
  sendChatMessage: (message: string) => void;
  roomId?: string;
  socket: Socket | null; // 
};

const GameContext = createContext<GameContextType | undefined>(undefined);

const GameProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [socket, setSocket] = useState<Socket | null>(null);
  const [gameState, setGameState] = useState<GameState | null>(null);
  const [isHost, setIsHost] = useState(false);
  const [chatMessages, setChatMessages] = useState<ChatMessage[]>([]);
  const [hasJoined, setHasJoined] = useState(false);
  const { user } = useAuth();
  const [alreadyHost, setAlreadyHost] = useState(false);
  const alreadyHostRef = useRef(false); // replaces useState for alreadyHost

  const waitForSocketConnection = (socket: Socket): Promise<void> =>
    new Promise((resolve) => {
      if (socket.connected) return resolve();
      socket.on('connect', () => resolve());
    });

  useEffect(() => {
    if (!user) return;
    const newSocket = io(window.location.hostname + ':5000');
    setSocket(newSocket);

    newSocket.on('connect', () => {
      console.log('[SOCKET] Connected:', newSocket.id);
    });

    newSocket.on('disconnect', () => {
      setHasJoined(false);
    });

    newSocket.on('roomCreated', (data) => {
  console.log('ROOM CREATED - setting host based on backend data');
  
  setGameState({
    ...data.roomData,
    availableMemes: MEMES,
    submissions: [],
    timeLeft: 0,
    gamePhase: 'lobby',
  });

  // Find current user in players and check isHost flag
  const me = data.roomData.players.find((p: Player) => p.id === user?.id);
  alreadyHostRef.current = me?.isHost === true;
  setIsHost(me?.isHost === true);
  setAlreadyHost(me?.isHost === true);
});


   newSocket.on('roomJoined', (data: { roomData: GameState }) => {
  console.log('Room joined data:', data);
  console.log('Current user ID:', user?.id);
  const me = data.roomData.players.find((p: Player) => p.id === user?.id);
  console.log('Matched player:', me);

 if (alreadyHostRef.current) {
  return console.warn('Already joined as host, ignoring room join event');
}

  console.log('ROOM JOINED ');

  setGameState({
    ...data.roomData,
    availableMemes: MEMES,
    submissions: data.roomData.submissions || [],
    timeLeft: 0,
    gamePhase: data.roomData.gamePhase || 'lobby',
  });

  if (user?.id === data.roomData.players[0]?.id) {
    setIsHost(true);
  }
});
  
newSocket.on('error', (data: { error: string }) => {
  console.error('[SOCKET ERROR]', data.error);
  alert(data.error);
  setGameState(null);
  setHasJoined(false);

  // Disconnect the socket so it doesn't keep retrying
  newSocket.disconnect();

  setTimeout(() => {
    window.location.href = '/join';
  }, 100);
});



    newSocket.on('playerJoined', (players) => {
  console.log('[SOCKET] playerJoined:', players);
  setGameState((prev) => prev ? { ...prev, players } : null);
});

newSocket.on('playerLeft', (updatedPlayers) => {
  console.log('[SOCKET] playerLeft:', updatedPlayers);
  setGameState((prev) => prev ? { ...prev, players: updatedPlayers } : null);
});

   newSocket.on('roomDiscarded', () => {
  console.log('roomDiscarded received');
  alert('Room has been discarded by the host.');
  setGameState(null);
  setIsHost(false);
  setChatMessages([]);
  setTimeout(() => {
    newSocket.disconnect();
    setSocket(null);
    window.location.href = '/dashboard';
  }, 200); // delay to ensure event processing
});


    newSocket.on('gamePhaseChanged', (newPhase) => {
      setGameState((prev) => (prev ? { ...prev, gamePhase: newPhase } : null));
    });

    newSocket.on('judgeSelected', (judge) => {
      setGameState((prev) =>
        prev
          ? {
              ...prev,
              currentJudge: judge,
              players: prev.players.map((p) => ({
                ...p,
                isJudge: p.id === judge.id,
              })),
            }
          : null
      );
    });

    newSocket.on('sentenceSubmitted', (sentence) => {
      setGameState((prev) => (prev ? { ...prev, currentSentence: sentence } : null));
    });

    newSocket.on('memeSelected', (submission) => {
      setGameState((prev) =>
        prev ? { ...prev, submissions: [...prev.submissions, submission] } : null
      );
    });

    newSocket.on('memeScored', ({ playerId, score }) => {
      setGameState((prev) => {
        if (!prev) return null;
        return {
          ...prev,
          submissions: prev.submissions.map((sub) =>
            sub.playerId === playerId ? { ...sub, score } : sub
          ),
          players: prev.players.map((p) =>
            p.id === playerId ? { ...p, score: p.score + score } : p
          ),
        };
      });
    });

    newSocket.on('chatMessage', (message) => {
      setChatMessages((prev) => [...prev, message]);
    });
  newSocket.on('gameStarted', (roomData) => {
  console.log('[SOCKET] Game started with data:', roomData);

  setGameState((prev) =>
    prev
      ? {
          ...prev,
          gamePhase: 'judgeSelection',
          currentJudge: roomData.currentJudge,
          currentRound: roomData.currentRound,
          players: roomData.players,
        }
      : null
  );

  window.location.href = '/game'; // redirect after backend confirms game start
});


     return () => {
    newSocket.disconnect();
    setSocket(null);
  };
}, [user]);

  const createRoom = async (settings: { rounds: number; roundsPerJudge: number }) => {
    if (!socket || !user) return '';

    return new Promise<string>((resolve, reject) => {
      socket.emit(
        'createRoom',
        {
          rounds: settings.rounds,
          roundsPerJudge: settings.roundsPerJudge,
          host: {
            id: user.id,
            username: user.username,
            avatar: user.avatar,
            score: 0,
            isJudge: false,
            isReady: true,
            isHost: true,
            // Host is always ready
          },
        },
        (response: { success: boolean; roomId?: string; error?: string }) => {
          if (response.success && response.roomId) {
            resolve(response.roomId);
          } else {
            reject(new Error(response.error || 'Room creation failed'));
          }
        }
      );
    });
  };

  const joinRoom = async (roomId: string) => {
    
    if (!socket || !user || hasJoined || alreadyHost) return;
    await waitForSocketConnection(socket);

    socket.emit('joinRoom', {
      roomId,
      player: {
        id: user.id,
        username: user.username,
        avatar: user.avatar,
        score: 0,
        isJudge: false,
        isReady: true,
      },
    });

    setHasJoined(true);
  };

  const leaveRoom = () => {
  if (socket && gameState?.roomId && user) {
    socket.emit('leaveRoom', { roomId: gameState.roomId, playerId: user.id });
    setGameState(null);
    setIsHost(false);
    setChatMessages([]);
    // Don't disconnect socket — player may rejoin another room later
    // Instead, just leave the room logically
    window.location.href = '/dashboard'; // or wherever you want to send them
  }
};
const discardRoom = () => {
  if (socket && gameState?.roomId) {
    socket.emit('discardRoom', { roomId: gameState.roomId });
    // Do not clean up here; wait for 'roomDiscarded' event
  }
};



  const startJudgeSelection = () => {
    if (socket && gameState) {
      socket.emit('startJudgeSelection', { roomId: gameState.roomId });
    }
  };

  const submitSentence = (sentence: string) => {
    if (socket && gameState) {
      socket.emit('submitSentence', {
        roomId: gameState.roomId,
        sentence,
      });
    }
  };

  const selectMeme = (memeId: string) => {
    if (socket && gameState && user) {
      socket.emit('selectMeme', {
        roomId: gameState.roomId,
        playerId: user.id,
        memeId,
      });
    }
  };

  const scoreMeme = (playerId: string, score: number) => {
    if (socket && gameState) {
      socket.emit('scoreMeme', {
        roomId: gameState.roomId,
        playerId,
        score,
      });
    }
  };

  const startNextRound = () => {
    if (socket && gameState) {
      socket.emit('startNextRound', { roomId: gameState.roomId });
    }
  };

  const endGame = () => {
    if (socket && gameState) {
      socket.emit('endGame', { roomId: gameState.roomId });
    }
    setGameState(null);
    setIsHost(false);
    setChatMessages([]);
  };

  const sendChatMessage = (message: string) => {
    if (socket && gameState && user) {
      socket.emit('chatMessage', {
        roomId: gameState.roomId,
        message: {
          id: Math.random().toString(),
          playerId: user.id,
          playerName: user.username,
          message,
          timestamp: new Date(),
        },
      });
    }
  };

  return (
    <GameContext.Provider
      value={{
        gameState,
        isHost,
        alreadyHost,
        setAlreadyHost,
        createRoom,
        joinRoom,
        leaveRoom,
        discardRoom, // ✅ now available to use
        startJudgeSelection,
        submitSentence,
        selectMeme,
        scoreMeme,
        startNextRound,
        endGame,
        chatMessages,
        sendChatMessage,
        socket,
        roomId: gameState?.roomId, // ✅ optional
      }}
    >
      {children}
    </GameContext.Provider>
  );
};

export const useGame = () => {
  const context = useContext(GameContext);
  if (context === undefined) {
    throw new Error('useGame must be used within a GameProvider');
  }
  return context;
};

export default GameProvider;
